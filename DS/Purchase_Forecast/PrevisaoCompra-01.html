Análise Exploratória dos Dados

Receita Por Loja

TOP 10

df |> group_by(loja) |> 
  summarise(valor_pago = sum(valor_pago)) |>
  slice_max(valor_pago, n=10) |> 
  ggplot(aes(x=fct_reorder(loja, valor_pago, .desc = TRUE), y=valor_pago, fill=loja)) + 
  geom_col()

BOTTOM 10
